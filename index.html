Each library or module plays a crucial role in web scraping, data processing, or visualization aspects of our project.

# helper function to get the city data from each link 
def get_city_data(url):
    driver.get(url)
    time.sleep(2)  
    city_data = {}
    
    # extracting data from the city link and getting the key and value correctly
    data = driver.find_elements(By.XPATH, "//div[@id='content']//table//tr")
    for i in data:
        element1 = i.find_elements(By.XPATH, ".//td[1]")
        element2 = i.find_elements(By.XPATH, ".//td[2]")
        if element1 and element2:
            key = element1[0].text.strip().rstrip(':')
            value = element2[0].text.strip()
            city_data[key] = value

    return city_data

def get_specific_data(row_data):
    data = {}
    
    # getting population and rank from the population row
    if pd.notna(row_data['Population']):
        population = row_data['Population'].split("rank #")
        data['Population'] = population[0].split(" ")[0]  
        data['Population Rank'] = "#" + population[1] 

    # getting income and rank from the median household income row
    if pd.notna(row_data['Median Household Income']):
        income = row_data['Median Household Income'].split("—")
        data['Median Household Income'] = income[0].split("at")[0]
        data['Income Rank'] = "#" + income[1].split("rank #")[1]

    # getting house price and ranking from median house price row
    if pd.notna(row_data['Median House Price']):
        house = row_data['Median House Price'].split("—")
        data['Median House Price'] = house[0].split("at")[0]
        data['House Price Rank'] = "#" + house[1].split("rank #")[1]

    # getting county and zipcode information
    data['County'] = row_data.get('County', None)
    data['Zip Code'] = row_data.get('Zip Code', None)

    result = pd.Series(data)
    
    return result

In this case, we assigned a weight of -1 to the crime index because the lower the crime rate, the safer the city is, and a weight of 1 to both median household income and median house price. Then, we computed a score for each city by multiplying each standardized factor by its respective weight and summing them up. This score represents the overall attractiveness of each city based on the selected criteria. At the end, we sort the cities based on their scores in descending order to identify the top 10 cities for us to consider living in Maryland.
